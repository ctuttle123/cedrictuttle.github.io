# Example: Buy-and-Hold daily returns
def buy_and_hold(df):
    df = df.copy()
    df['Signal'] = 1  # Always invested
    df['Strategy_Return'] = df['Return'] * df['Signal']
    df['Cumulative_Return'] = (1 + df['Strategy_Return']).cumprod()
    return df

# Selected tickers
tickers = ['AAPL', 'MSFT', 'JNJ', 'JPM', 'XOM', 'KO', 'DIS', 'CVX']

# Apply Buy-and-Hold to all tickers and store in a dictionary
bh_dict = {}
for ticker in tickers:
    bh_dict[ticker] = buy_and_hold(stocks[stocks['Ticker'] == ticker])

# Example: Momentum strategy based on 20-day MA
def momentum_strategy(df):
    df = df.copy()
    # Buy when price is above MA20, otherwise cash (Signal = 0)
    df['Signal'] = (df['Close'] > df['MA20']).astype(int)
    df['Strategy_Return'] = df['Return'] * df['Signal']
    df['Cumulative_Return'] = (1 + df['Strategy_Return']).cumprod()
    return df

# Selected tickers
tickers = ['AAPL', 'MSFT', 'JNJ', 'JPM', 'XOM', 'KO', 'DIS', 'CVX']

# Apply Momentum strategy to all tickers and store in a dictionary
mom_dict = {}
for ticker in tickers:
    mom_dict[ticker] = momentum_strategy(stocks[stocks['Ticker'] == ticker])

def pretty_metrics(avg_returns, avg_cumulative, strategy_name):
    mean_return = avg_returns.mean()
    volatility = avg_returns.std()
    sharpe_ratio = mean_return / volatility * (252**0.5)  # Annualized
    cumulative_return = avg_cumulative.iloc[-1]
    
    # Print nicely
    print(f"--- {strategy_name} Metrics ---")
    print(f"Mean Daily Return: {mean_return*100:.3f}%")
    print(f"Daily Volatility: {volatility*100:.3f}%")
    print(f"Annualized Sharpe Ratio: {sharpe_ratio:.2f}")
    print(f"Cumulative Return: {cumulative_return:.2f}x")
    print()

ml_dfs = []

for ticker in tickers:
    df = stocks[stocks['Ticker'] == ticker].copy()
    
    # Lagged returns
    for lag in range(1, 6):
        df[f'Return_Lag{lag}'] = df['Return'].shift(lag)
    
    # Target: next-day positive return
    df['Target'] = (df['Return'].shift(-1) > 0).astype(int)
    
    # Drop NaNs
    df = df.dropna().reset_index(drop=True)
    
    # Keep feature columns + Ticker
    df = df[['Ticker', 'Date', 'MA20', 'Vol20'] + [f'Return_Lag{i}' for i in range(1,6)] + ['Target']]
    
    ml_dfs.append(df)

# Combine all tickers
ml_data = pd.concat(ml_dfs, ignore_index=True)

# Feature columns
feature_cols = [f'Return_Lag{i}' for i in range(1,6)] + ['MA20', 'Vol20']
target_col = 'Target'

ML Strategy Performance and Limitations
The ML-driven strategy we implemented using a Random Forest classifier did not outperform the simpler strategies (Buy-and-Hold and Momentum) in terms of cumulative return or Sharpe ratio. Key reasons include:


    

  1. Low predictive accuracy (~51%)
    

      

    • The model predicts next-day returns barely better than random chance.

  2. High noise in daily stock returns
    

      

    • Stock prices fluctuate in ways that are largely random in the short term, making next-day prediction extremely difficult.
      • 

    

    3. 

  3. Limited features
    

      

    • We only used 5 lagged returns, 20-day moving average, and 20-day volatility.
      • 

    • Additional indicators (longer-term trends, volume patterns, sector info) could improve predictions.
      • 

    

    4. 

  4. Model choice & overfitting risk

    5. 



Takeaway:
